"""Intentionally vulnerable authentication module (academic use only)."""

import sqlite3
from pathlib import Path

DB = str(Path(__file__).resolve().parent / "users.db")

# Intentionally vulnerable by design for controlled coursework demonstration.
# Never use this pattern in real systems.


def vulnerable_login(username: str, password: str):
	"""Authenticate via unsafe SQL string concatenation.

	This function is purposely insecure so the assignment can demonstrate
	SQL injection-based authentication bypass behavior.
	"""

	query = (
		"SELECT * FROM users "
		"WHERE username = '" + username + "' "
		"AND password_plain = '" + password + "'"
	)

	conn = sqlite3.connect(DB)
	conn.row_factory = sqlite3.Row
	cursor = conn.cursor()

	try:
		cursor.execute(query)
		user = cursor.fetchone()
	except Exception as e:
		conn.close()
		return None, query, str(e)

	conn.close()

	if user:
		return dict(user), query, None
	return None, query, None
